// Form submission routes for contact and complaints
Route::post('/contact', function (Illuminate\Http\Request $request) {
    // Handle contact form submission
    $request->validate([
        'name' => 'required|string|max:255',
        'email' => 'required|email|max:255',
        'subject' => 'required|string|max:255',
        'message' => 'required|string|max:2000',
        'phone' => 'nullable|string|max:20'
    ]);

    // Here you would typically save to database or send email
    // For now, just return success
    return redirect()->back()->with('success', 'تم إرسال رسالتك بنجاح. سنتواصل معك قريباً');
})->name('contact.store');

Route::post('/complaints', function (Illuminate\Http\Request $request) {
    // Handle complaints form submission
    $request->validate([
        'name' => 'required|string|max:255',
        'email' => 'required|email|max:255',
        'type' => 'required|string|in:complaint,suggestion,bug,feedback',
        'subject' => 'required|string|max:255',
        'description' => 'required|string|max:2000',
        'phone' => 'nullable|string|max:20',
        'priority' => 'nullable|string|in:low,medium,high,urgent',
        'ad_id' => 'nullable|string|max:50'
    ]);

    // Here you would typically save to database
    // For now, just return success
    return redirect()->back()->with('success', 'تم إرسال شكواك/اقتراحك بنجاح. سنراجعه ونرد عليك قريباً');
})->name('complaints.store');
